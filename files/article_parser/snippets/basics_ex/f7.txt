import math

def is_prime(x):
  if x == 2 or x == 3:
    return True

  if x % 2 == 0:
    return False

  for i in xrange(3, int(math.sqrt(x) + 1), 2):
    if x % i == 0:
      return False

  return True

sum = 0
count = 0

limit = 1000000

for i in xrange(2, limit):
  if is_prime(i):
    sum += i
    count += 1

print "The sum of the %s primes between 2 and %s is %s." % (count, limit, sum)
